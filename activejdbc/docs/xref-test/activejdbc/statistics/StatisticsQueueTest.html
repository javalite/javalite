20  import static javalite.test.jspec.JSpec.a;
21  import org.junit.Test;

28  public class StatisticsQueueTest {
29  
30      @Test
31      public void shouldCollectAndSortStatistics(){
32  
33          StatisticsQueue queue = new StatisticsQueue();
34          queue.start();
35  
36          for(int i = 0; i < 10; i++){
37              queue.enqueue(new QueryExecutionEvent("test", 10 + i));
38          }
39  
40          for(int i = 0; i < 10; i++){
41              queue.enqueue(new QueryExecutionEvent("test1", 20 + i));
42          }
43  
44          for(int i = 0; i < 10; i++){
45              queue.enqueue(new QueryExecutionEvent("test2", 30 + i));
46          }
47  
48          //could be a race condition, let's wait till all messages are processed
49          try{Thread.sleep(1000);}catch(Exception e){/* ignore*/}
50  
51          List<QueryStats> report = queue.getReportSortedBy("avg");
52  
53          a(report.get(0).getQuery()).shouldBeEqual("test2");
54          a(report.get(0).getAvg()).shouldBeEqual(35);
55          a(report.get(0).getTotal()).shouldBeEqual(345);
56          a(report.get(0).getCount()).shouldBeEqual(10);
57          a(report.get(0).getMin()).shouldBeEqual(30);
58          a(report.get(0).getMax()).shouldBeEqual(39);
59  
60  
61          a(report.get(1).getQuery()).shouldBeEqual("test1");
62          a(report.get(1).getAvg()).shouldBeEqual(25);
63          a(report.get(1).getTotal()).shouldBeEqual(245);
64          a(report.get(1).getCount()).shouldBeEqual(10);
65          a(report.get(1).getMin()).shouldBeEqual(20);
66          a(report.get(1).getMax()).shouldBeEqual(29);
67          //TODO: add checks of all values - total, avg, count, etc.
68  
69  
70          a(report.get(2).getQuery()).shouldBeEqual("test");
71          a(report.get(2).getAvg()).shouldBeEqual(15);
72          a(report.get(2).getTotal()).shouldBeEqual(145);
73          a(report.get(2).getCount()).shouldBeEqual(10);
74          a(report.get(2).getMin()).shouldBeEqual(10);
75          a(report.get(2).getMax()).shouldBeEqual(19);
76  
77          queue.stop();
78          try{Thread.sleep(1000);}catch(Exception e){/* ignore*/} //wait till the queue stops
79      }
